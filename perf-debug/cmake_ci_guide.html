2.1 иҖҒејҸ CMake (CMake 2.x ж—¶д»Ј)

# иҖҒејҸ - directory-based, е…ЁеұҖеҸҳйҮҸж»ЎеӨ©йЈһ

5.1 дёүдёӘе…ій”®еӯ—зҡ„еҗ«д№ү

  2. A зҡ„еӨҙж–Үд»¶жҡҙйңІдәҶ X еҗ—? (й“ҫжҺҘ A зҡ„дәәиғҪзңӢеҲ° X)
A иҮӘз”Ё?еӨҙж–Үд»¶жҡҙйңІ?з”Ё

function(set_project_warnings target_name)
    set(MSVC_WARNINGS
        /W4 /permissive- /w14242 /w14254 /w14263 /w14265 /w14287
        /w14289 /w14296 /w14311 /w14545 /w14546 /w14547 /w14549
        /w14555 /w14619 /w14640 /w14826 /w14905 /w14906 /w14928
    )
    set(CLANG_WARNINGS
        -Wall -Wextra -Wpedantic
        -Wshadow -Wnon-virtual-dtor
        -Wold-style-cast -Wcast-align -Wunused
        -Woverloaded-virtual -Wconversion -Wsign-conversion
        -Wdouble-promotion -Wformat=2
        -Wimplicit-fallthrough
    )
    set(GCC_WARNINGS
        ${CLANG_WARNINGS}
        -Wmisleading-indentation -Wduplicated-cond
        -Wduplicated-branches -Wlogical-op -Wuseless-cast
    )

    if(MSVC)
        set(WARNINGS ${MSVC_WARNINGS})

        set(WARNINGS ${CLANG_WARNINGS})
    elseif(CMAKE_CXX_COMPILER_ID STREQUAL "GNU")
        set(WARNINGS ${GCC_WARNINGS})
    endif()

    target_compile_options(${target_name} PRIVATE ${WARNINGS})
endfunction()

# cmake/Sanitizers.cmake
function(enable_sanitizers target)
    if(NOT (CMAKE_CXX_COMPILER_ID STREQUAL "GNU" OR
            CMAKE_CXX_COMPILER_ID MATCHES ".*Clang"))
        return()
    endif()

    set(SANITIZERS "")
    if(ENABLE_ASAN)
        list(APPEND SANITIZERS "address")
    endif()
    if(ENABLE_UBSAN)
        list(APPEND SANITIZERS "undefined")
    endif()
    if(ENABLE_TSAN)
        if(ENABLE_ASAN OR ENABLE_MSAN)
            message(FATAL_ERROR "TSan cannot combine with ASan/MSan")
        endif()
        list(APPEND SANITIZERS "thread")
    endif()
    if(ENABLE_MSAN)
        list(APPEND SANITIZERS "memory")
    endif()

    list(JOIN SANITIZERS "," SAN_LIST)
    if(SAN_LIST)
        target_compile_options(${target} PRIVATE
            -fsanitize=${SAN_LIST}
            -fno-omit-frame-pointer
            -fno-optimize-sibling-calls
        )
        target_link_options(${target} PRIVATE -fsanitize=${SAN_LIST})
    endif()
endfunction()
