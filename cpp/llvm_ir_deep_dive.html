target triple = "x86_64-pc-linux-gnu"

define dso_local noundef i32 @_Z3addii(i32 noundef %0, i32 noundef %1) #0 {
  %3 = alloca i32, align 4              ; еұҖйғЁ var a (ж Ҳ)
  %4 = alloca i32, align 4              ; еұҖйғЁ var b
  %5 = alloca i32, align 4              ; иҝ”еӣһеҖј slot
  store i32 %0, ptr %3, align 4
  store i32 %1, ptr %4, align 4
  %6 = load i32, ptr %3, align 4
  %7 = load i32, ptr %4, align 4
  %8 = icmp sgt i32 %6, %7              ; a > b
  br i1 %8, label %9, label %11

9:                                       ; preds = %2
  %10 = load i32, ptr %3
  store i32 %10, ptr %5
  br label %13

11:                                      ; preds = %2
  %12 = load i32, ptr %4
  store i32 %12, ptr %5
  br label %13

13:                                      ; preds = %11, %9
  %14 = load i32, ptr %5
  ret i32 %14
}

define noundef i32 @_Z3addii(i32 noundef %a, i32 noundef %b) {
entry:

03

SSA дёҺ Phi иҠӮзӮ№

й—®йўҳпјҡif-else жҖҺд№Ҳ SSAпјҹ В· SSA дёҚиғҪе®Ңе…ЁиЎЁиҫҫд»Җд№Ҳ

SSA (Static Single Assignment)вҖ”вҖ”жҜҸдёӘеҸҳйҮҸеҸӘиөӢеҖјдёҖж¬ЎгҖӮеҗ¬иө·жқҘйҷҗеҲ¶еӨ§пјҢдҪҶе®һйҷ…и®© def-use еҲҶжһҗ trivialвҖ”вҖ”жҜҸдёӘ use еҸӘиғҪеҜ№еә”дёҖдёӘ defгҖӮ

й—®йўҳпјҡif-else жҖҺд№Ҳ SSAпјҹ

int x;
if (cond) x = 1;
else    x = 2;
use(x);    // x жңү 2 дёӘ defпјҢиҝқеҸҚ SSA

и§Јпјҡphi иҠӮзӮ№вҖ”вҖ”"ж №жҚ®иө°зҡ„е“ӘжқЎеҲҶж”ҜжҢ‘йҖүеҗҲйҖӮзҡ„ def"пјҡ

entry:
  %cond = ...
  br i1 %cond, label %if_true, label %if_false

if_true:
  %x1 = i32 1
  br label %merge

if_false:
  %x2 = i32 2
  br label %merge

merge:
  %x = phi i32 [ %x1, %if_true ], [ %x2, %if_false ]
  call void @use(i32 %x)

phi иҠӮзӮ№еңЁ basic block йЎ¶йғЁпјҢж №жҚ®"д»Һе“ӘжқЎ incoming edge жқҘ"йҖүдёҚеҗҢ defгҖӮиҝҷжҳҜ SSA зҡ„ж ёеҝғеҲӣж–°гҖӮ

SSA дёҚиғҪе®Ңе…ЁиЎЁиҫҫд»Җд№Ҳ

  • йҖғйҖёеҲ°еҶ…еӯҳзҡ„еҜ№иұЎвҖ”вҖ”иҝҳжҳҜ alloca + load/storeпјҲдёҚиғҪ mem2regпјү
  • йқһеұҖйғЁи·ЁеҮҪж•° defвҖ”вҖ”global еҸҳйҮҸ / heap еҜ№иұЎ
  • еҫӘзҺҜеҪ’зәіеҸҳйҮҸвҖ”вҖ”йңҖиҰҒ phi иҠӮзӮ№й…Қ induction variable analysis

жүҖд»Ҙ SSA жҳҜеҮҪж•°еҶ…зҡ„пјҢи·ЁеҮҪж•°д»ҚйңҖ inter-procedural analysisгҖӮ

зұ»еһӢзі»з»ҹ

еҹәзЎҖзұ»еһӢ В· иҒҡеҗҲзұ»еһӢ В· Opaque Pointers (LLVM 16+)

LLVM IR жҳҜејәзұ»еһӢвҖ”вҖ”дёҚеғҸдј з»ҹжұҮзј–гҖӮжҜҸдёӘ value йғҪжңү typeпјҢoperation еҝ…йЎ» type-compatibleгҖӮ

iNпјҲд»»ж„Ҹ Nпјүд»»ж„Ҹ bit widthпјҲеҰӮ i7пјүеҒ¶е°”з”ЁдәҺ bit-packed ж•°жҚ®

Opaque Pointers (LLVM 16+)

2022 е№ҙиө· LLVM е…ЁйғЁ pointer зұ»еһӢ typelessвҖ”вҖ”д№ӢеүҚжҳҜ i32* / %struct.X* / ptrпјҢзҺ°еңЁз»ҹдёҖеҸӘ ptrгҖӮеҺҹеӣ пјҡ

  • еҺҶеҸІ typed pointer жҳҜ design mistakeпјҲtype и·ҹ pointer ејәз»‘еҜјиҮҙ bitcast ж»ЎеӨ©йЈһпјү
  • е®һйҷ…жҢҮй’Ҳзҡ„"type"еә”иҜҘеңЁ load/store/GEP жҢҮд»ӨдёҠжҳҫејҸж Ү
  • з®ҖеҢ– IR + еҮҸе°‘еҶ—дҪҷдҝЎжҒҜ
; иҖҒејҸ (LLVM < 15):
%1 = load i32, i32* %ptr_to_int
%2 = load %struct.X, %struct.X* %ptr_to_struct

; Opaque pointers (LLVM 16+):
%1 = load i32, ptr %ptr_any
%2 = load %struct.X, ptr %ptr_any
                  // вҶ‘ зұ»еһӢдҝЎжҒҜеңЁ load жҢҮд»Өжң¬иә«

еҪұе“ҚпјҡеҶҷ LLVM pass дёҚиғҪеҶҚдҫқиө– pointer's "pointee type"пјҢеҝ…йЎ»зңӢ use siteгҖӮ

дёӘе…ій”®дјҳеҢ– Pass еҶ…йғЁ

Inliner вҖ” еҮҪж•°дҪ“жӣҝд»Ј call В· mem2reg вҖ” жҠҠж ҲеҸҳйҮҸжҸҗеҚҮеҲ° SSAпјҲи§Ғ В§3пјү В· GVN (Global Value Numbering) вҖ” иҜҶеҲ«зӯүд»·иЎЁиҫҫејҸ В· LICM (Loop InvarвҖҰ

7.1 Inliner вҖ” еҮҪж•°дҪ“жӣҝд»Ј call

  • ж ёеҝғз®—жі•пјҡcost-benefitгҖӮCost = inlined еҗҺзҡ„д»Јз Ғ size еўһй•ҝпјӣBenefit = ж¶ҲйҷӨ call overhead + жҡҙйңІеҗҺз»ӯдјҳеҢ–
  • и°ғз”ЁеӣҫиҮӘеә•еҗ‘дёҠвҖ”вҖ”е…Ҳ inline leaf functionпјҢеҶҚ inline и°ғз”Ёе®ғд»¬зҡ„
  • еҗҜеҸ‘ејҸпјҡеҮҪж•°е°ҸпјҲвүӨ ~15 IR instrпјү/ caller еҶ…зҡ„ constant arg / hot call site вҶ’ inline
  • е…ёеһӢж”¶зӣҠпјҡ5-20%гҖӮд№ҹжҳҜжүҖжңүеҗҺз»ӯдјҳеҢ–зҡ„еүҚжҸҗ

7.2 mem2reg вҖ” жҠҠж ҲеҸҳйҮҸжҸҗеҚҮеҲ° SSAпјҲи§Ғ В§3пјү

7.3 GVN (Global Value Numbering) вҖ” иҜҶеҲ«зӯүд»·иЎЁиҫҫејҸ

; Before GVN:
%a = add i32 %x, %y
%b = add i32 %x, %y      ; зӯүд»·иЎЁиҫҫејҸ
%c = mul i32 %a, %b

; After GVN:
%a = add i32 %x, %y
%c = mul i32 %a, %a      ; %b иў«ж¶ҲйҷӨ

з»“еҗҲ MemorySSA + AliasAnalysis еҒҡ load eliminationгҖӮж”¶зӣҠпјҡе…ёеһӢ hot path 1-5%гҖӮ

7.4 LICM (Loop Invariant Code Motion)

; Before:
for (i = 0; i < n; i++) {
  x = a * b;          // invariant in loop
  arr[i] = x + i;
}

; After LICM:
x = a * b;            // hoisted out
for (i = 0; i < n; i++) {
  arr[i] = x + i;
}

йңҖиҰҒ LoopInfo + AliasAnalysisгҖӮж”¶зӣҠпјҡеҫӘзҺҜеҜҶйӣҶд»Јз Ғ 10-30%гҖӮ

7.5 SROA (Scalar Replacement of Aggregates)

жҠҠиҒҡеҗҲпјҲstruct/arrayпјүжӢҶжҲҗеҚ•зӢ¬ scalarгҖӮй…ҚеҗҲ mem2reg и®© struct field д№ҹиғҪиҝӣ registerгҖӮжһҒйҮҚиҰҒ passвҖ”вҖ”жІЎ SROA зҺ°д»Ј C++ жҖ§иғҪеҙ©гҖӮ

7.6 Loop Vectorizer вҖ” SIMD иҮӘеҠЁе№¶иЎҢеҢ–

; Before:
for (i = 0; i < 1000; i++)
  c[i] = a[i] + b[i];

; After vectorize (AVX2):
for (i = 0; i < 1000; i += 8) {
  <8 x float> va = load <8 x float>, a+i
  <8 x float> vb = load <8 x float>, b+i
  <8 x float> vc = fadd va, vb
  store <8 x float> vc, c+i
}

йңҖиҰҒ iteration count е·ІзҹҘ / ж—  dependency between iterations / еҜ№йҪҗ OKгҖӮnoalias и®© vectorize жӣҙжҝҖиҝӣгҖӮж”¶зӣҠпјҡж•°жҚ®еҜҶйӣҶ loop 2-8Г—гҖӮ

7.7 InstCombine вҖ” еұҖйғЁжЁЎејҸйҮҚеҶҷ

%t = mul i32 %x, 2          ; вҶ’ %t = shl i32 %x, 1
%t = mul i32 %x, 8          ; вҶ’ %t = shl i32 %x, 3
%t = and i32 %x, 0xffff     ; bit mask дјҳеҢ–
%t = sub i32 %x, %x         ; вҶ’ 0
%t = add i32 %x, 0          ; вҶ’ %x

InstCombine и·‘еҫҲеӨҡж¬ЎпјҲжҜҸдёӘ transform pass д№ӢеҗҺпјүпјҢжҢҒз»ӯжё…зҗҶ IRгҖӮзҙҜи®Ўж”¶зӣҠдёҚеӨ§дҪҶ ubiquitousгҖӮ

еҠЁжүӢеҶҷдёҖдёӘ LLVM pass жҳҜзңҹжӯЈзҗҶи§Ј LLVM зҡ„ fastest wayгҖӮдёӢйқўжҳҜе®Ңж•ҙзҡ„"з»ҹи®ЎеҮҪж•°дёӯзҡ„ call жҢҮд»Өж•°йҮҸ" passгҖӮ

MyPass.cppпјҲж ёеҝғпјү

#include "llvm/IR/Function.h"
#include "llvm/IR/PassManager.h"
#include "llvm/Passes/PassPlugin.h"

using namespace llvm;

struct CallCounterPass : public PassInfoMixin<CallCounterPass> {
  PreservedAnalyses run(Function &F, FunctionAnalysisManager &) {
    unsigned count = 0;
    for (BasicBlock &BB : F)
      for (Instruction &I : BB)
        if (isa<CallInst>(&I)) ++count;

    errs() << "Function " << F.getName()
           << " has " << count << " call instructions\n";
    return PreservedAnalyses::all();
  }
};

// Plugin entry
extern "C" ::llvm::PassPluginLibraryInfo LLVM_ATTRIBUTE_WEAK
llvmGetPassPluginInfo() {
  return { LLVM_PLUGIN_API_VERSION, "CallCounter", "1.0",
    [](PassBuilder &PB) {
      PB.registerPipelineParsingCallback(
        [](StringRef Name, FunctionPassManager &FPM, auto) {
          if (Name == "call-counter") {
            FPM.addPass(CallCounterPass());
            return true;
          }
          return false;
        });
    }};
}

# Build

# Run йҖҡиҝҮ opt

    -passes="call-counter" -disable-output test.ll

# иҫ“еҮә:
# Function main has 3 call instructions

жҸ’жЎ©пјҲmodifying IRпјүзҡ„ pass

PreservedAnalyses run(Function &F, FunctionAnalysisManager &) {
  auto *M = F.getParent();
  auto &Ctx = F.getContext();
  FunctionCallee LogFn = M->getOrInsertFunction(
    "__my_log",
    FunctionType::get(Type::getVoidTy(Ctx),
                      {PointerType::get(Ctx, 0)}, false));

  for (auto &BB : F)
    for (auto &I : BB)
      if (auto *CI = dyn_cast<CallInst>(&I)) {
        IRBuilder<> B(CI);
        Value *Name = B.CreateGlobalStringPtr(
          CI->getCalledFunction()->getName());
        B.CreateCall(LogFn, {Name});  // log еңЁ call д№ӢеүҚ
      }
  return PreservedAnalyses::none();
}

иҝҷе°ұжҳҜ sanitizer / coverage tool / profiler зҡ„ж ёеҝғжңәеҲ¶вҖ”вҖ”еңЁ LLVM IR дёҠжҸ’жЎ©гҖӮ
