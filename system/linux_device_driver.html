01

驱动在哪：用户态 vs 内核态

你写的 open/read 最终落到谁身上

在用户态你调用 open("/dev/mydev")、read(fd, buf, n)、ioctl(fd, CMD, arg)。驱动就是内核里接住这些调用的一段代码：它把「一次系统调用」翻译成「对某个硬件寄存器的读写 / 对某块内存的搬运」，再把结果交回用户态。换句话说，文件描述符那篇讲的 struct file::f_op 函数表，表里的函数指针正是驱动要填的坑。

生活类比：餐厅的后厨与传菜口

用户态程序是顾客，只能在传菜口（系统调用界面）点单：给菜名（fd）和要求（参数）。后厨（内核）不让顾客进——怕碰坏设备、也怕看到别桌的菜（安全隔离）。驱动就是后厨里对接某台设备（烤箱 / 冰箱）的专职厨师：顾客说"烤箱调 200 度"（ioctl），厨师去按烤箱的物理旋钮（写寄存器）。顾客永远碰不到旋钮，只能通过传菜口下指令。

映射：顾客=用户进程 · 传菜口=syscall / VFS · 厨师=驱动 · 物理旋钮=MMIO 寄存器 · 不让进后厨=用户/内核地址空间隔离

1.1 内核态代码的四条铁律

进入内核态写代码，用户态的很多"常识"立刻失效。这不是风格问题，而是硬约束：

约束为什么后果 / 替代
没有 libc / STL / printf内核不链接用户态 C 库用 printk、kmalloc、内核链表 list_head
栈极小（通常 8KB/16KB）每线程内核栈固定，且不可增长禁止大数组 / 深递归，大缓冲用 kmalloc
不能随意用浮点 / SIMD内核默认不保存 FPU 上下文需要时显式 kernel_fpu_begin/end
一个 bug = 整机崩没有进程边界保护你空指针 → oops / panic，而不是 SIGSEGV
名词：内核空间 / 用户空间指针

内核代码里裸指针分两种世界：内核指针（内核直接解引用）和用户指针（标注 __user，绝不能直接解引用）。用户传进来的地址可能非法、可能被并发换出、可能是攻击构造的——必须走 copy_from_user() 这类带检查的搬运函数。这是内核安全的第一道闸。

1.2 一次 read 的完整旅程

read 从用户态陷入内核，VFS 靠 f_op 函数表把调用分发到你的驱动，驱动读硬件后 copy_to_user 回填
真正要理解的是：驱动不是"程序"，而是一堆"回调"。内核在需要时调用你注册的函数（open/read/write/ioctl/中断处理），你几乎从不主动"运行"。写驱动的心智模型是事件驱动 + 被内核调用，不是 main() 里的顺序流程。
02

第一个内核模块

insmod / rmmod / dmesg 的最小闭环

驱动通常以可加载内核模块（LKM, .ko）的形式存在：编译成独立文件，运行时 insmod 装进内核、rmmod 卸载，不用重编内核、不用重启。这是嵌入式调试的效率关键。

// hello.c — 最小内核模块
#include <linux/module.h>
#include <linux/init.h>

static int __init hello_init(void)
{
    pr_info("hello: loaded, minor base=%d\n", 0);
    return 0;   // 返回非 0 → 加载失败，模块不驻留
}

static void __exit hello_exit(void)
{
    pr_info("hello: unloaded\n");
}

module_init(hello_init);   // insmod 时调用
module_exit(hello_exit);   // rmmod 时调用

MODULE_LICENSE("GPL");      // 非 GPL 会 taint 内核，且用不了很多导出符号
MODULE_AUTHOR("you");
MODULE_DESCRIPTION("minimal LKM");

2.1 Kbuild：内核的构建规则

模块不能用普通 gcc hello.c 编译——它必须针对特定内核版本的头文件和配置构建，靠内核源码树的 Kbuild 系统。

# Makefile
obj-m += hello.o                 # 目标：编成模块（-m），不是内建（-y）

# 本机构建：指向已装的内核头
KDIR ?= /lib/modules/$(shell uname -r)/build

all:
	$(MAKE) -C $(KDIR) M=$(PWD) modules
clean:
	$(MAKE) -C $(KDIR) M=$(PWD) clean
# 加载 / 查看 / 卸载的最小闭环
make
sudo insmod hello.ko
dmesg | tail            # 看到 "hello: loaded"
lsmod | grep hello
sudo rmmod hello
dmesg | tail            # 看到 "hello: unloaded"
名词：交叉编译（Cross-Compile）

嵌入式几乎从不在目标板上编译。你在 x86 主机上用 ARM 工具链编出 ARM 的 .ko：make ARCH=arm64 CROSS_COMPILE=aarch64-linux-gnu- -C $KDIR M=$PWD modules，其中 $KDIR 指向目标板同版本的内核源码/头。版本或配置（vermagic）对不上，insmod 会直接拒绝加载。工程细节见 Modern CMake 的 Cross-Compile。

pr_info/pr_err 是 printk(KERN_INFO ...) 的封装。内核没有 stdout——所有输出都进内核日志环形缓冲区，用 dmesg 或 /dev/kmsg 看。日志级别低于 console_loglevel 才会打到串口/控制台，这也是嵌入式看不到 log 时的头号排查点。
03

字符设备三件套

设备号 + cdev + file_operations

Linux 设备分三类：字符设备（按字节流访问，如串口、传感器、GPIO——绝大多数嵌入式外设）、块设备（按块随机访问，如 eMMC/SSD）、网络设备（不走 /dev，走 socket）。本文聚焦最常写的字符设备。让 /dev/mydev 能 open/read/write，需要三样东西对齐。

3.1 三件套的关系

设备号是身份，cdev 是内核对象，fops 是行为；三者注册后 udev 在 /dev 建出用户可见节点

3.2 完整的字符设备骨架

#include <linux/module.h>
#include <linux/fs.h>        // file_operations, register_chrdev_region
#include <linux/cdev.h>
#include <linux/uaccess.h>   // copy_to/from_user
#include <linux/device.h>    // class_create → udev 建节点

#define BUF_SZ 1024

struct mydev {
    struct cdev cdev;
    char buf[BUF_SZ];
    size_t len;
    struct mutex lock;      // 保护 buf/len，见第 7 章
};

static dev_t         dev_num;      // major:minor
static struct class *cls;
static struct mydev  *g;

static int mydev_open(struct inode *ino, struct file *f)
{
    // container_of：从内嵌的 cdev 反推出外层 struct mydev
    f->private_data = container_of(ino->i_cdev, struct mydev, cdev);
    return 0;
}

static ssize_t mydev_read(struct file *f, char __user *ubuf,
                          size_t n, loff_t *ppos)
{
    struct mydev *d = f->private_data;
    ssize_t ret;

    mutex_lock(&d->lock);
    if (*ppos >= d->len) { ret = 0; goto out; }   // EOF
    n = min_t(size_t, n, d->len - *ppos);
    if (copy_to_user(ubuf, d->buf + *ppos, n)) { ret = -EFAULT; goto out; }
    *ppos += n;
    ret = n;
out:
    mutex_unlock(&d->lock);
    return ret;
}

static ssize_t mydev_write(struct file *f, const char __user *ubuf,
                           size_t n, loff_t *ppos)
{
    struct mydev *d = f->private_data;
    n = min_t(size_t, n, BUF_SZ);
    mutex_lock(&d->lock);
    if (copy_from_user(d->buf, ubuf, n)) { mutex_unlock(&d->lock); return -EFAULT; }
    d->len = n;
    mutex_unlock(&d->lock);
    return n;
}

static const struct file_operations fops = {
    .owner = THIS_MODULE,          // 防止使用中被 rmmod（refcount）
    .open  = mydev_open,
    .read  = mydev_read,
    .write = mydev_write,
    .llseek = default_llseek,
};

3.3 注册流程与 udev 建节点

static int __init mydev_init(void)
{
    int ret;
    // 1) 动态申请一个设备号（major 由内核分配，别写死！）
    ret = alloc_chrdev_region(&dev_num, /*baseminor*/0, /*count*/1, "mydev");
    if (ret) return ret;

    g = kzalloc(sizeof(*g), GFP_KERNEL);
    if (!g) { ret = -ENOMEM; goto err_region; }
    mutex_init(&g->lock);

    // 2) 初始化 cdev 并挂上 fops，加入内核
    cdev_init(&g->cdev, &fops);
    g->cdev.owner = THIS_MODULE;
    ret = cdev_add(&g->cdev, dev_num, 1);
    if (ret) goto err_free;

    // 3) 建 class + device → udev 自动创建 /dev/mydev
    cls = class_create("mydev_class");
    if (IS_ERR(cls)) { ret = PTR_ERR(cls); goto err_cdev; }
    device_create(cls, NULL, dev_num, NULL, "mydev");

    pr_info("mydev: major=%d minor=%d\n", MAJOR(dev_num), MINOR(dev_num));
    return 0;

// 错误路径逆序回滚 —— 内核编程的必修课，漏一步就泄漏
err_cdev:  cdev_del(&g->cdev);
err_free:  kfree(g);
err_region:unregister_chrdev_region(dev_num, 1);
    return ret;
}

static void __exit mydev_exit(void)
{
    device_destroy(cls, dev_num);
    class_destroy(cls);
    cdev_del(&g->cdev);
    kfree(g);
    unregister_chrdev_region(dev_num, 1);
}
module_init(mydev_init);
module_exit(mydev_exit);
MODULE_LICENSE("GPL");
解释 container_of(ino->i_cdev, struct mydev, cdev)。它是内核实现"面向对象"的核心宏：已知内嵌成员 cdev 的地址，用编译期 offsetof 反算出外层结构体 struct mydev 的首地址。这样你可以把私有状态（缓冲、锁、寄存器基址）挂在 struct mydev 里，从任何回调拿到它。不用全局变量、支持多实例的关键就是它 + file::private_data。
动态申请设备号（alloc_chrdev_region）而不是写死 major，是现代驱动的规矩：写死的 major 会和别的驱动冲突。用户态需要的映射交给 udev / /proc/devices 解决。
04

与用户态的数据通道

copy_*_user / ioctl / mmap 三种流派

驱动和用户态之间搬数据有三条路，性能和用途各不相同。选错会要么慢、要么不安全。

通道方向 / 语义开销适用
read/write + copy_*_user字节流，有拷贝每次一次内核↔用户拷贝常规数据流（串口、传感器采样）
ioctl带外控制命令 + 小结构体一次拷贝小数据配置寄存器、查询状态、模式切换
mmap把设备内存/缓冲映射进用户地址空间建立映射后零拷贝帧缓冲、大块 DMA 缓冲、高吞吐

4.1 copy_to_user / copy_from_user：为什么不能直接 memcpy

用户指针永远不可信。直接 memcpy(kbuf, ubuf, n) 会：① 若 ubuf 非法 → 内核 oops（而不是杀进程）；② 被恶意构造成内核地址 → 越权读写（经典提权漏洞）；③ 该页可能被换出，需要缺页处理。copy_*_user 做地址范围检查（access_ok）、处理缺页、失败返回"未拷贝字节数"。

4.2 ioctl：设备的"控制面"

read/write 是数据面，ioctl 是控制面。命令号必须用内核宏编码"方向 + 幻数 + 序号 + 参数大小"，避免不同驱动命令撞车：

#include <linux/ioctl.h>
#define MYDEV_MAGIC  'k'
#define MYDEV_RESET     _IO(MYDEV_MAGIC,   0)                // 无参
#define MYDEV_SET_FREQ  _IOW(MYDEV_MAGIC,  1, int)           // 用户→内核
#define MYDEV_GET_STATS _IOR(MYDEV_MAGIC,  2, struct stats) // 内核→用户

static long mydev_ioctl(struct file *f, unsigned int cmd, unsigned long arg)
{
    struct mydev *d = f->private_data;
    switch (cmd) {
    case MYDEV_RESET:
        mutex_lock(&d->lock); d->len = 0; mutex_unlock(&d->lock);
        return 0;
    case MYDEV_SET_FREQ: {
        int hz;
        if (get_user(hz, (int __user *)arg)) return -EFAULT;  // 小标量用 get_user
        if (hz < 1 || hz > 100000) return -EINVAL;            // 永远校验入参
        /* writel(hz, d->regs + FREQ); */
        return 0;
    }
    default:
        return -ENOTTY;   // 约定：未知 ioctl 返回 -ENOTTY（不是 -EINVAL）
    }
}
.unlocked_ioctl（不是老的 .ioctl）是现代接口。64 位内核跑 32 位用户程序时，结构体布局可能不同，需要额外实现 .compat_ioctl 做转换。命令用 _IO/_IOR/_IOW/_IOWR 宏编码大小，能在早期就拦住"参数结构体不匹配"的调用。

4.3 mmap：零拷贝的代价与用途

当吞吐是瓶颈（摄像头帧、GPU、采集卡），逐字节 copy_to_user 太贵。mmap 直接把设备/内核缓冲映射进用户地址空间，之后用户读写就是访存，无系统调用：

static int mydev_mmap(struct file *f, struct vm_area_struct *vma)
{
    struct mydev *d = f->private_data;
    unsigned long pfn;

    // 设备寄存器区：禁用 cache（否则读到旧值）
    vma->vm_page_prot = pgprot_noncached(vma->vm_page_prot);
    pfn = d->phys_base >> PAGE_SHIFT;   // 物理地址 → 页帧号

    return remap_pfn_range(vma, vma->vm_start, pfn,
                           vma->vm_end - vma->vm_start,
                           vma->vm_page_prot);
}
映射设备寄存器一定要用 pgprot_noncached（或 write-combining）。若走普通可缓存映射，CPU 会把对寄存器的写攒在 cache 里、把读命中旧值——硬件行为完全错乱。这类"cache 一致性"陷阱在第 8 章 DMA 会再次出现，是嵌入式最难缠的一类 bug。
05

设备树与 platform_driver

别再把板级信息写死进 C 代码

x86 有 PCI/ACPI 能自动枚举硬件，但 ARM/RISC-V SoC 上大量外设（I²C 控制器、SPI、GPIO、UART、传感器）挂在片上总线，硬件无法自我描述——寄存器基址、中断号、时钟在哪，CPU 不知道。历史上这些信息硬编码在内核的 "board file" 里，导致每块板子都要改内核、几千个 board file 让 ARM 内核树臃肿不堪（Linus 当年为此发过火）。设备树（Device Tree）就是把这些描述从代码里抽出来，变成一份独立的数据文件。

生活类比：装修图纸 vs 把家具焊死在墙上

旧做法（board file）像把家具焊死在墙上：换个户型就得砸墙重焊（改内核重编）。设备树像一份可替换的装修图纸：墙体（内核 + 驱动）不动，只换图纸（.dtb）就能描述"这块板子上，UART 在地址 0x..、用中断 32、接 24MHz 时钟"。同一个内核镜像配不同 dtb 就能跑不同板子。

映射：墙体=内核/驱动逻辑 · 图纸=设备树 dts/dtb · 家具位置=reg/interrupts/clocks · 换户型不砸墙=同内核多板复用

5.1 设备树节点长什么样

// board.dts —— 描述一个自定义外设
myperiph@40001000 {
    compatible = "acme,myperiph-v1";      // ← 匹配键，驱动靠它认领
    reg = <0x40001000 0x1000>;           // 寄存器基址 + 长度
    interrupts = <GIC_SPI 42 IRQ_TYPE_LEVEL_HIGH>;  // 中断号 + 触发类型
    clocks = <&clk_periph>;               // 时钟源引用
    acme,sample-rate-hz = <48000>;        // 自定义属性（厂商前缀）
    status = "okay";
};

5.2 platform_driver：probe 时拿到这些资源

内核用 compatible 字符串把「设备树节点」和「驱动」配对；配上了就调用驱动的 probe()，并把资源交给它
#include <linux/platform_device.h>
#include <linux/of.h>
#include <linux/io.h>
#include <linux/interrupt.h>

struct myperiph {
    void __iomem *regs;     // 映射后的寄存器基址
    int irq;
    u32 sample_rate;
};

static int myperiph_probe(struct platform_device *pdev)
{
    struct device *dev = &pdev->dev;
    struct myperiph *p;

    p = devm_kzalloc(dev, sizeof(*p), GFP_KERNEL);  // devm_：自动随设备释放
    if (!p) return -ENOMEM;

    // 1) 拿寄存器：devm_platform_ioremap_resource 读 dts 的 reg 并 ioremap
    p->regs = devm_platform_ioremap_resource(pdev, 0);
    if (IS_ERR(p->regs)) return PTR_ERR(p->regs);

    // 2) 拿中断号（dts 的 interrupts）
    p->irq = platform_get_irq(pdev, 0);
    if (p->irq < 0) return p->irq;

    // 3) 读自定义属性
    of_property_read_u32(dev->of_node, "acme,sample-rate-hz", &p->sample_rate);

    // 4) 申请中断（见第 6 章），设备就绪
    int ret = devm_request_irq(dev, p->irq, myperiph_isr, 0, "myperiph", p);
    if (ret) return ret;

    platform_set_drvdata(pdev, p);
    dev_info(dev, "probed, rate=%u irq=%d\n", p->sample_rate, p->irq);
    return 0;
}

static const struct of_device_id myperiph_of[] = {
    { .compatible = "acme,myperiph-v1" },
    { }
};
MODULE_DEVICE_TABLE(of, myperiph_of);

static struct platform_driver myperiph_drv = {
    .probe  = myperiph_probe,
    .driver = { .name = "myperiph", .of_match_table = myperiph_of },
};
module_platform_driver(myperiph_drv);   // 展开 init/exit，省样板
解释 devm_* 家族（managed device resources）。devm_kzalloc/devm_request_irq/devm_ioremap 分配的资源绑定到 device 生命周期，设备移除或 probe 失败时内核自动逆序释放。它把第 3 章那种"手写 goto 回滚"消灭掉了，是现代驱动减少泄漏的头号实践。但注意：devm 的释放时机是"设备解绑时"，不适合需要在运行中反复申请/释放的资源。
面试常问"设备树解决了什么问题"：答案不是"配置硬件"，而是把易变的板级描述与稳定的驱动逻辑解耦，使一个内核镜像 + 多份 dtb 支持无数板子，终结了 ARM 内核里泛滥的 board file。
06

中断：上半部与下半部

为什么中断处理要"拆两半"

硬件事件（数据到了、传输完成、按键按下）通过中断异步通知 CPU。CPU 立刻暂停当前工作，跳去执行中断服务程序（ISR）。问题在于：中断处理期间通常关着（同优先级）中断，处理越久，系统对其它事件越迟钝、延迟抖动越大。所以内核把中断处理拆成两半。

名词：中断上下文（interrupt context）

ISR 运行在"中断上下文"里——不属于任何进程、不能睡眠（不能调 mutex_lock、kmalloc(GFP_KERNEL)、copy_to_user 等可能阻塞的函数）。因为没有可切换回的进程上下文，一旦睡眠就死锁/崩溃。这是内核编程最容易踩的红线。

6.1 上半部 / 下半部分工

上半部只做"必须马上做"的事（应答硬件），把耗时工作丢给下半部；要睡眠就用 workqueue（进程上下文）

6.2 下半部三种机制怎么选

机制上下文能否睡眠并发特性用途
softirq中断上下文否同类型可在多核并行网络/块层等高频路径（内核内部用）
tasklet中断上下文否同一 tasklet 不会并行普通驱动下半部（正被 threaded IRQ 取代）
workqueue进程上下文是内核线程执行，可调度需要睡眠/分配/加锁的耗时活
threaded IRQ内核线程是每中断一线程，PREEMPT_RT 友好现代首选，见下

6.3 现代写法：threaded IRQ

与其手动管理 tasklet/workqueue，现代驱动更常用线程化中断：一个函数做上半部（quick check），一个函数做下半部（跑在专属内核线程，可以睡眠）。这也是 PREEMPT_RT 让中断可抢占、压低延迟的基础。

// 上半部：中断上下文，只判断"是不是我的中断"，尽快返回
static irqreturn_t myperiph_isr(int irq, void *data)
{
    struct myperiph *p = data;
    u32 st = readl(p->regs + REG_STATUS);
    if (!(st & INT_PENDING))
        return IRQ_NONE;              // 不是我的（共享中断线时很关键）

    writel(INT_PENDING, p->regs + REG_ACK);  // 应答，防止反复触发
    return IRQ_WAKE_THREAD;           // 唤醒下半部线程
}

// 下半部：内核线程上下文，可以睡眠 / kmalloc / 拿 mutex / 唤醒等待队列
static irqreturn_t myperiph_thread(int irq, void *data)
{
    struct myperiph *p = data;
    // ... 搬运/解析数据、写入缓冲、wake_up_interruptible(&p->wq) ...
    return IRQ_HANDLED;
}

// 在 probe 里注册（threaded 版本）
devm_request_threaded_irq(dev, p->irq,
        myperiph_isr,        // 上半部（可为 NULL，全丢给线程）
        myperiph_thread,     // 下半部
        IRQF_ONESHOT, "myperiph", p);
上半部里绝对不能：mutex_lock、msleep、kmalloc(GFP_KERNEL)（会睡）、copy_to_user。需要这些就放到下半部（线程/workqueue）。要在中断和进程上下文之间共享数据，只能用会关中断的自旋锁 spin_lock_irqsave——见下一章。
07

内核并发与竞态

选错锁 = 死锁或数据损坏

内核天生高度并发：多核同时进你的驱动、中断随时打断进程上下文、抢占让内核代码中途被换出。共享数据（缓冲、计数、寄存器影子值）必须保护。选锁的第一问永远是：持锁期间会不会睡眠？会不会被中断访问同一数据？

7.1 mutex vs spinlock 决策

mutex（可睡眠锁）

  • 拿不到锁 → 睡眠让出 CPU
  • 只能用在进程上下文
  • 临界区可以睡眠 / 长
  • ❌ 中断上下文里用 = 崩溃

spinlock（自旋锁）

  • 拿不到锁 → 忙等自旋（不睡）
  • 可用于中断上下文
  • 临界区必须极短、不能睡眠
  • 单核编译时退化为仅关抢占
口诀：临界区会睡就用 mutex；不能睡（或中断里）就用 spinlock，且临界区越短越好。自旋锁持有期间睡眠是内核头号 bug——会让持锁者被换出、别的核死等，直接 hang。

7.2 与中断共享数据：spin_lock_irqsave

若同一份数据既被进程上下文（read/write/ioctl）又被中断处理访问，只用普通 spinlock 不够：进程持锁时若被本核中断打断、中断里又试图拿同一锁 → 自死锁。必须用会关本核中断的版本：

unsigned long flags;

// 进程上下文 & 中断处理都用这一对，保护同一份数据
spin_lock_irqsave(&p->slock, flags);   // 关本核中断 + 上锁，保存原中断状态
/* 极短临界区：改共享环形缓冲的头尾指针 */
spin_unlock_irqrestore(&p->slock, flags); // 解锁 + 恢复中断状态
场景用哪把锁
只在进程上下文、临界区可能睡眠mutex
只在进程上下文、临界区极短不睡spin_lock / spin_lock_bh
数据被中断处理也访问spin_lock_irqsave
读多写极少RCU / seqlock（读者近乎无锁）
纯计数器atomic_t / per-CPU 变量
性能进阶：热点路径别无脑加大锁。内核提供 per-CPU 变量（每核一份，免同步）、RCU（读者无锁、写者延迟释放，见 并发深入 里的 RCU/Hazard Pointer）。这些是把驱动做到高吞吐、低抖动的关键，思路和用户态 lock-free 一脉相承。

7.3 内存屏障：编译器和 CPU 都会重排

操作 MMIO 寄存器时，写的顺序常有硬件含义（先写地址寄存器、再写触发寄存器）。编译器优化和 CPU 乱序会打乱它。readl/writel 已含必要屏障，但裸内存共享（尤其和 DMA）要显式 wmb()/rmb()/mb()、smp_wmb() 等。这与用户态 C++ 的 内存模型 / memory order 是同一套底层规律的两种表达。

08

DMA 与一致性内存

让外设自己搬数据，但小心 cache

DMA（直接内存访问）让外设绕过 CPU 直接读写内存：网卡收包、摄像头写帧、磁盘控制器搬扇区，都靠它。CPU 只需"告诉外设缓冲在哪、多大"，然后去干别的，传输完外设发中断。它是高吞吐外设的命脉，但引出两个硬核问题：地址与cache 一致性。

名词：总线地址 vs 物理地址 vs 虚拟地址

CPU 用虚拟地址，经 MMU 转成物理地址。但外设看到的是总线/DMA 地址——在有 IOMMU 或地址偏移的系统上，它和物理地址并不相等。所以不能把 kmalloc 得到的内核虚拟地址直接给外设，必须用 DMA API 拿到正确的 dma_addr_t。

8.1 一致性 DMA vs 流式 DMA

一致性 (coherent) 映射

  • dma_alloc_coherent()
  • 缓冲区免手动同步（不可缓存或硬件保证一致）
  • 适合：环形描述符、控制块、双方频繁读写
  • 代价：可能较慢（禁 cache）

流式 (streaming) 映射

  • dma_map_single() / dma_unmap_single()
  • 用完即映射，需手动 sync + 声明方向
  • 适合：一次性大块传输（网络包、磁盘 IO）
  • 性能好，但用错方向 = 数据损坏

8.2 cache 一致性问题的本质

CPU 读写内存走 cache。DMA 直接改的是物理内存。于是：① 外设写完内存，CPU 若命中旧的 cache 行 → 读到脏数据；② CPU 写了数据还留在 cache 没回写，外设读内存 → 读到旧值。没有硬件 cache 一致性的平台（很多 ARM SoC），必须靠软件在恰当时机 invalidate / flush cache——这正是流式 DMA 要 dma_sync_* 的原因。
// 一致性映射：适合外设与 CPU 都要频繁访问的描述符环
p->desc = dma_alloc_coherent(dev, RING_BYTES, &p->desc_dma, GFP_KERNEL);
// p->desc      : CPU 用的虚拟地址
// p->desc_dma  : 交给外设的 DMA 地址
writel(lower_32_bits(p->desc_dma), p->regs + REG_RING_BASE);

// 流式映射：一次性发送一大块，声明方向后手动管理
dma_addr_t da = dma_map_single(dev, tx_buf, len, DMA_TO_DEVICE);
if (dma_mapping_error(dev, da)) return -ENOMEM;
writel(da, p->regs + REG_TX_ADDR);
writel(len, p->regs + REG_TX_LEN);
// ... 传输完成中断到来后 ...
dma_unmap_single(dev, da, len, DMA_TO_DEVICE);  // 此时才安全复用 tx_buf
DMA 方向（DMA_TO_DEVICE / DMA_FROM_DEVICE / DMA_BIDIRECTIONAL）不是文档注释，而是告诉内核在何时 flush 还是 invalidate cache。写反了，在无一致性硬件的平台上会读到脏数据——而且往往只在特定 cache 状态下偶发，是最难复现的 bug 之一。

调试与观测

没有 gdb 断点时你还有什么

内核态没有 printf 调试那么随意，一个空指针可能直接 panic。好在工具链很成熟：

手段用途要点
printk / dev_dbg打点日志受 console_loglevel 影响；dev_dbg 靠 dynamic debug 动态开关
dmesg / oops 解析崩溃后看栈回溯用 addr2line / faddr2line 把地址还原成源码行
ftrace / trace_printk函数调用图、延迟分析低开销，能在中断上下文用，见 eBPF / perf
sysfs / debugfs暴露驱动内部状态给用户态调试用 debugfs，正式 ABI 用 sysfs（要写文档）
KGDB / KASAN源码级调试 / 越界检测KASAN 是内核版 ASan，抓 use-after-free / 越界

9.1 用 sysfs 暴露一个可读写属性

// cat /sys/.../sample_rate 读，echo N > ... 写
static ssize_t sample_rate_show(struct device *dev,
        struct device_attribute *a, char *buf)
{
    struct myperiph *p = dev_get_drvdata(dev);
    return sysfs_emit(buf, "%u\n", p->sample_rate);  // sysfs_emit 防溢出
}
static ssize_t sample_rate_store(struct device *dev,
        struct device_attribute *a, const char *buf, size_t n)
{
    struct myperiph *p = dev_get_drvdata(dev);
    u32 v;
    if (kstrtou32(buf, 0, &v)) return -EINVAL;
    p->sample_rate = v;
    return n;
}
static DEVICE_ATTR_RW(sample_rate);   // 生成 dev_attr_sample_rate
调试内核 hang / 死锁时，别只盯日志：echo l > /proc/sysrq-trigger（打所有 CPU 栈）、lockdep（编译期开 CONFIG_PROVE_LOCKING 自动检测锁序死锁）、ftrace 的 function graph 看卡在哪个函数。嵌入式常配一根串口控制台，panic 时的信息只能从它出来。
10

高频面试题

中高级嵌入式 / 内核岗常考
中断处理函数里为什么不能睡眠？如果非要做耗时/阻塞的活怎么办？

中断上下文不绑定任何进程，没有可切回的 task_struct，也就没有"稍后被唤醒继续"的载体；一旦睡眠，调度器无法恢复它，轻则该 CPU hang、重则 panic。而且上半部通常关着中断，睡在里面会拉高全系统中断延迟。做法：上半部只应答硬件、置标志、尽快返回；耗时/阻塞工作交给下半部——需要睡眠就用 workqueue 或 threaded IRQ（跑在内核线程，可睡眠、可 kmalloc(GFP_KERNEL)、可拿 mutex）。

spinlock 和 mutex 怎么选？spin_lock 和 spin_lock_irqsave 的区别？

选择第一问是"临界区会不会睡眠"：会睡（或临界区长）→ mutex（拿不到锁时让出 CPU）；不能睡或在中断上下文 → spinlock（忙等，临界区必须极短）。spin_lock 只防其它核；若这份数据也被中断处理访问，进程上下文持普通 spinlock 时被本核中断打断、中断里再拿同一锁会自死锁，必须用 spin_lock_irqsave（额外关本核中断并保存/恢复中断状态）。

设备树（Device Tree）解决了什么问题？compatible、reg、interrupts 分别是什么？

它把易变的板级硬件描述从内核代码里剥离成独立数据文件，让一个内核镜像配不同 dtb 支持不同板子，终结了 ARM 内核里泛滥的 board file。compatible 是匹配键，内核用它把设备树节点和驱动的 of_match_table 配对，配上就调 probe()；reg 是寄存器基址+长度（驱动 ioremap 用）；interrupts 是中断号和触发类型（platform_get_irq 取）。

copy_to_user 和 memcpy 有什么区别？为什么不能直接解引用用户指针？

用户指针可能非法、可能指向已被换出的页、可能被恶意构造成内核地址。直接 memcpy 会 oops 或造成越权读写（经典提权面）。copy_*_user 会 access_ok 校验地址范围属于用户空间、能处理缺页、失败返回未拷贝字节数。小标量用 get_user/put_user。这是用户/内核隔离的强制闸门。

DMA 的 cache 一致性问题是什么？一致性映射和流式映射怎么选？

CPU 走 cache、DMA 直改物理内存，二者可能看到不同版本：外设写完内存 CPU 命中旧 cache 行读到脏数据，或 CPU 写还在 cache 没回写外设读到旧值。无硬件一致性的平台要软件 flush/invalidate。一致性映射（dma_alloc_coherent）免手动同步，适合双方频繁访问的描述符环；流式映射（dma_map_single）性能好但要声明方向并手动 dma_sync/unmap，适合一次性大块传输。方向写反在无一致性平台会偶发数据损坏。

container_of 是怎么工作的？为什么驱动里到处都是它？

已知结构体某个内嵌成员的地址，用编译期 offsetof(type, member) 减去偏移，反算出外层结构体首地址。驱动把私有状态（缓冲、锁、寄存器基址）内嵌在自定义结构里，再把内核对象（cdev/work_struct/timer_list）作为成员；回调只拿到内核对象指针时，靠 container_of 找回私有数据，从而不用全局变量、天然支持多实例。

.owner = THIS_MODULE 有什么用？不写会怎样？

它让内核在设备被打开/使用时对模块加引用计数，阻止使用中的模块被 rmmod。不写的话，正有进程 open 着 /dev/mydev 时模块被卸载，回调函数所在代码被释放 → 再调用即野指针崩溃。

11

checklist 与小结

写字符 + 平台驱动的自检清单

  • 设备号动态申请（alloc_chrdev_region），别写死 major。
  • 私有状态挂 struct + container_of / private_data，不用全局变量、支持多实例。
  • 资源优先用 devm_*，probe 失败/移除自动回滚；手写路径必须逆序释放。
  • 用户指针一律 copy_*_user / get_user；ioctl 入参严格校验，未知命令返回 -ENOTTY。
  • 板级信息走设备树，compatible 匹配，reg/interrupts/clocks 从 dts 取。
  • 中断拆上半部/下半部：上半部快而不睡，耗时/阻塞放 threaded IRQ 或 workqueue。
  • 共享数据选对锁：睡→mutex，不睡→spinlock，被中断访问→spin_lock_irqsave，临界区极短。
  • DMA 用 DMA API 拿 dma_addr_t；流式映射声明方向并 sync，注意 cache 一致性。
  • MMIO / mmap 设备寄存器用 pgprot_noncached；跨设备/DMA 共享加内存屏障。
  • 调试备好串口控制台 + dmesg/ftrace；开发期开 KASAN + lockdep。
一句话总结：驱动是内核里被"回调"的一堆函数——把 fops 填对、把资源用 devm 管好、把中断拆成两半、把锁和 cache 一致性想清楚，一个能上真实 SoC 的驱动就成了。
