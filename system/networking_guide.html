#include <sys/socket.h>
#include <netinet/in.h>
#include <unistd.h>
#include <cstring>
#include <iostream>

int main() {
    // 1. 创建 socket (IPv4, TCP)
    int listenfd = socket(AF_INET, SOCK_STREAM, 0);

    // 2. 绑定地址和端口
    sockaddr_in addr{};
    addr.sin_family = AF_INET;
    addr.sin_addr.s_addr = htonl(INADDR_ANY);  // 监听所有网卡
    addr.sin_port = htons(8080);               // 端口 8080

    // SO_REUSEADDR: 允许快速重启时复用端口
    int opt = 1;
    setsockopt(listenfd, SOL_SOCKET, SO_REUSEADDR, &opt, sizeof(opt));

    bind(listenfd, (sockaddr*)&addr, sizeof(addr));

    // 3. 开始监听，backlog = 128
    listen(listenfd, 128);
    std::cout << "Server listening on port 8080...\n";

    // 4. 接受连接 (阻塞)
    sockaddr_in client_addr{};
    socklen_t client_len = sizeof(client_addr);
    int connfd = accept(listenfd, (sockaddr*)&client_addr, &client_len);

    // 5. 读写数据
    char buf[1024];
    ssize_t n = read(connfd, buf, sizeof(buf));
    buf[n] = '\0';
    std::cout << "Received: " << buf << "\n";

    const char* resp = "Hello from server!";
    write(connfd, resp, strlen(resp));

    // 6. 关闭
    close(connfd);
    close(listenfd);
}

3.3 长度前缀协议实现代码

// 发送端: 先发长度，再发数据
bool sendMessage(int fd, const std::string& msg) {
    uint32_t len = htonl(msg.size());  // 网络字节序!
    // 发送 4 字节长度头
    if (write(fd, &len, 4) != 4) return false;
    // 发送消息体
    if (write(fd, msg.data(), msg.size()) != (ssize_t)msg.size()) return false;
    return true;
}

// 接收端: 先读长度，再读精确数量的字节
std::string recvMessage(int fd) {
    // 1. 读取 4 字节长度头
    uint32_t netLen;
    readExact(fd, &netLen, 4);         // 必须精确读满 4 字节
    uint32_t len = ntohl(netLen);

    // 2. 根据长度读取消息体
    std::string msg(len, '\0');
    readExact(fd, msg.data(), len);    // 必须精确读满 len 字节
    return msg;
}

// 关键辅助函数: 确保读满 n 字节 (处理拆包)
void readExact(int fd, void* buf, size_t n) {
    size_t totalRead = 0;
    while (totalRead < n) {
        ssize_t r = read(fd, (char*)buf + totalRead, n - totalRead);
        if (r <= 0) throw std::runtime_error("connection closed");
        totalRead += r;
    }
}

5.2 epoll 核心 API

#include <sys/epoll.h>

// 1. 创建 epoll 实例
int epfd = epoll_create1(0);

// 2. 注册感兴趣的 fd 和事件
struct epoll_event ev;
ev.events = EPOLLIN | EPOLLET;  // 读事件 + 边缘触发
ev.data.fd = listenfd;
epoll_ctl(epfd, EPOLL_CTL_ADD, listenfd, &ev);

// 3. 事件循环
struct epoll_event events[1024];
while (true) {
    int n = epoll_wait(epfd, events, 1024, -1);  // 阻塞等待

    for (int i = 0; i < n; ++i) {
        if (events[i].data.fd == listenfd) {
            // 新连接到来
            int connfd = accept(listenfd, ...);
            // 设置非阻塞 + 注册到 epoll
            setNonBlocking(connfd);
            ev.events = EPOLLIN | EPOLLET;
            ev.data.fd = connfd;
            epoll_ctl(epfd, EPOLL_CTL_ADD, connfd, &ev);
        } else {
            // 数据到来，处理读写
            handleConnection(events[i].data.fd);
        }
    }
}

零拷贝 (Zero-Copy) 技术减少文件传输过程中 CPU 参与的数据拷贝次数，从而大幅提升文件传输 / 静态资源服务的性能。

6.1 传统文件传输 vs sendfile

6.2 sendfile 代码

#include <sys/sendfile.h>
#include <fcntl.h>

// 传统方式: read + write (4次拷贝)
char buf[8192];
while ((ssize_t n = read(filefd, buf, sizeof(buf))) > 0) {
    write(sockfd, buf, n);  // 数据经过用户空间中转
}

// sendfile 方式: 零拷贝 (2次DMA拷贝)
struct stat st;
fstat(filefd, &st);
sendfile(sockfd,    // 目标: socket fd
         filefd,    // 源: 文件 fd
         nullptr,  // offset (nullptr = 从当前位置)
         st.st_size // 发送的字节数
);
// 数据直接从文件 → 内核 → 网卡，不经过用户空间!

7b. 现代协议 — QUIC / HTTP/3 / TLS 1.3 / WebSocket Senior

7b.1 TLS 1.3 —— 握手砍半的密钥协商 · 7b.2 QUIC —— TCP 的"用户态 + UDP"替代 · 7b.3 HTTP/3 —— QUIC 上的 HTTP · 7b.4 WebSocket / SSE — 长连接…

TCP + TLS 1.2 + HTTP/2 已经统治网络十年，但队头阻塞（HOL blocking）、握手 RTT 多、移动场景切网卡断连三大老问题催生了下一代：QUIC + HTTP/3 + TLS 1.3。Cloudflare / Google / Meta 已普遍采用，C++ 工程师必须能讲清楚。

7b.1 TLS 1.3 —— 握手砍半的密钥协商

对比 TLS 1.2 vs 1.3

项TLS 1.2TLS 1.3 (2018)
建连 RTT2 RTT1 RTT（首次）/ 0 RTT（恢复）
密钥交换RSA / DH / ECDH 多选仅 (EC)DHE — 强制 PFS
加密算法20+ cipher suite5 个 AEAD only（AES-GCM / ChaCha20-Poly1305）

HKDF—统一的 key schedule
消息加密从 Finished 后ServerHello 后立即加密

0-RTT 重连（恢复会话）

// 客户端缓存了 session ticket + 早期数据
ssl_set_session(ssl, cached_session);
SSL_write_early_data(ssl, request_data, len);   // ★ 第 0 个 RTT 就发应用数据!
// 服务端拒绝则降级为普通 1-RTT
0-RTT 风险：重放攻击——攻击者重放 0-RTT 包多次执行同一请求。所以 0-RTT 数据必须是幂等的（GET / HEAD），写操作不能用。CDN 边缘节点对此做严格 enum 控制。

C++ 实战（OpenSSL 1.1.1+）

SSL_CTX_set_min_proto_version(ctx, TLS1_3_VERSION);     // 强制 1.3 only

    "TLS_AES_128_GCM_SHA256:TLS_CHACHA20_POLY1305_SHA256");
SSL_CTX_set_max_early_data(ctx, 16384);                 // 允许 0-RTT

7b.2 QUIC —— TCP 的"用户态 + UDP"替代

QUIC（RFC 9000，原 Google 2012）跑在 UDP 上，把可靠传输 + 拥塞控制 + 加密都搬到用户态。终结 TCP/TLS 的几大顽疾。

QUIC 解决了什么

TCP/TLS 痛点QUIC 解法
建连 1 RTT TCP + 1-2 RTT TLS1 RTT 建连 + 加密，0-RTT 恢复
HTTP/2 队头阻塞（一个流的丢包阻塞所有流）每流独立可靠，丢包只影响该流
切换网络（WiFi → 4G）TCP 必须重连Connection ID（与 IP 无关）→ 连接迁移
TCP 拥塞算法在内核（升级慢）拥塞算法在用户态 → 应用层可独立部署 BBR
中间盒子改协议（HTTP 升级 HTTPS 困难）整个 QUIC 包除少数字段全加密，盒子无法干涉

sudo sysctl -w net.core.default_qdisc=fq         # BBR 强烈推荐 fq
sudo sysctl -w net.ipv4.tcp_congestion_control=bbr

# 持久化
echo "net.core.default_qdisc=fq" >> /etc/sysctl.conf
echo "net.ipv4.tcp_congestion_control=bbr" >> /etc/sysctl.conf

# 看哪些算法可选
sysctl net.ipv4.tcp_available_congestion_control
实测收益（公开数据）：Google YouTube 视频缓冲减少 20-30%，Dropbox CDN 下载提速 30-50%（亚太、印度更显著）。但 BBR vs CUBIC 共存时不公平（BBR 会抢更多带宽），同机房内部建议统一选择。

7c.2 SO_REUSEPORT —— 多进程/多线程监听同一端口

Linux 3.9+ 的杀手特性。允许多个进程/线程同时 bind 同一 (IP, port)，内核做4 元组哈希分流到各 listener，彻底解决 nginx 旧版 "thundering herd" 问题。

替代旧的 master-fork-worker 模式

// 每个 worker 进程 / 线程独立 socket + bind
for (int i = 0; i < N_WORKERS; ++i) {
    if (fork() == 0) {
        int sock = socket(AF_INET, SOCK_STREAM, 0);
        int opt = 1;
        setsockopt(sock, SOL_SOCKET, SO_REUSEPORT, &opt, sizeof(opt));   // ★
        bind(sock, ...);
        listen(sock, 128);
        worker_loop(sock);     // 内核自动分流连接
        break;
    }
}

对比传统模式

项master + accept + work-stealingSO_REUSEPORT 多 listener
accept 锁竞争有（一个 accept 队列）无（每 listener 独立队列）
负载均衡worker 抢内核 4-tuple hash 自动
worker 升级 / 平滑重启需 socket fd 传递新进程直接 bind 同端口
每连接 cache 友好差（worker 随机）★★★★ 同一 4-tuple 永远去同 listener
实战：nginx 1.9.1+ 用 listen 80 reuseport; 启用；Envoy / HAProxy 默认开。Cloudflare 内部用它做"每核独立 accept"，单机能处理 10M+ 并发连接。

7c.3 TCP FastOpen — 把数据塞进握手包

普通 TCP 必须先 3 次握手才能发数据，TFO（RFC 7413）允许客户端在 SYN 包里就携带应用数据，节省 1 RTT。

// 服务端启用
int qlen = 5;
setsockopt(sock, IPPROTO_TCP, TCP_FASTOPEN, &qlen, sizeof(qlen));

// 客户端发起：sendto 替代 connect+send
sendto(sock, data, len, MSG_FASTOPEN, (sockaddr*)&addr, sizeof(addr));

需要：① Linux 3.7+；② 双端都支持；③ HTTP/2 / gRPC 等长连接收益小，短连接（DNS / HTTP/1.0）收益大。

7c.4 TCP_NODELAY / TCP_CORK / TCP_QUICKACK 三件套

option作用用在哪
TCP_NODELAY禁用 Nagle 算法（小包不缓存等大包合并）低延时 / RPC / 交互式
TCP_CORK反 Nagle —— 强制缓存直到 cork 解除或 buffer 满批量发大块（如 sendfile + header）
TCP_QUICKACK禁用延迟 ACK（200ms 等待）请求-响应短交互
常见坑：Nagle + delayed ACK 互相等待 → 出现 200ms "卡顿"。一般 TCP_NODELAY 总该开（除非你确实想合并小包）。

7c.5 SO_BUSY_POLL — 用户态主动轮询

把 NIC 中断 → 内核 → 应用通知的"等中断"流程改成应用主动 poll，消灭中断延迟。延迟敏感场景能从 ~10 μs 降到 ~3 μs。

// 应用 socket 上启用
int us = 50;        // 最多忙等 50 μs
setsockopt(sock, SOL_SOCKET, SO_BUSY_POLL, &us, sizeof(us));

// 也可以全局
sysctl -w net.core.busy_poll=50
sysctl -w net.core.busy_read=50

代价：吃 CPU 高（轮询时本核 100%）。HFT / 低延时 RPC 用，普通业务不要开。

面试高频问题速答

为什么 TCP 握手是三次而不是两次？

两次握手无法确认客户端的接收能力。若客户端的旧 SYN 延迟到达，服务端会错误建立连接（已失效的连接请求）。三次握手确保双方都确认了对方的收发能力。

为什么挥手要四次而不是三次？

TCP 是全双工的。收到 FIN 只是对方不再发送，但自己可能还有数据要发。所以 ACK 和自己的 FIN 是分开发送的（中间可能还要传数据）。

TIME_WAIT 状态为什么要等 2MSL？

① 确保最后的 ACK 能到达对端（若丢失，对端会重发 FIN）。② 等待网络中该连接的残余数据包消失，防止新连接收到旧数据。MSL = Maximum Segment Lifetime ≈ 30-60秒。

epoll 的 ET 模式为什么要配合非阻塞 fd？

ET 模式只在状态变化时通知一次。如果用阻塞 fd，read 到最后没数据时会阻塞住线程，无法处理其他 fd。必须用非阻塞 fd + 循环读直到 EAGAIN。

9.1 TCP 拥塞控制 — 网络流量的平衡艺术

TCP 通过拥塞窗口 cwnd（发送端）和接收窗口 rwnd（接收端）动态调整发送速率。实际发送量 = min(cwnd, rwnd)。

阶段条件cwnd 变化
慢启动cwnd < ssthresh每 ACK +1 MSS（指数增长）
拥塞避免cwnd ≥ ssthresh每 RTT +1 MSS（线性）
快重传收到 3 个重复 ACK立即重传，不等超时
快恢复快重传后cwnd = cwnd/2，ssthresh = cwnd/2
超时RTO 到期cwnd=1，ssthresh=cwnd/2，回到慢启动

现代算法：Reno → Cubic（Linux 默认）→ BBR（Google，基于带宽×延迟，不以丢包为信号）。查看：sysctl net.ipv4.tcp_congestion_control

9.2 Nagle + 延迟 ACK 的交互陷阱

Nagle 算法：小包（< MSS）延迟发送，累积成大包再发。
延迟 ACK：收到数据后不立即回 ACK，等 40ms 或有数据捎带。
两者叠加：请求-响应型应用会出现 ~40ms 莫名延迟。

// 关闭 Nagle — 实时协议的标配
int flag = 1;
setsockopt(fd, IPPROTO_TCP, TCP_NODELAY, &flag, sizeof(flag));

// 关闭延迟 ACK (Linux)
setsockopt(fd, IPPROTO_TCP, TCP_QUICKACK, &flag, sizeof(flag));

// 或：应用层 buffer，writev 一次发完整消息 (最优)
writev(fd, iov, 2);  // header + body 原子写

原则：请求-响应（RPC、RPC）一律 TCP_NODELAY；吞吐型（文件传输）保留 Nagle。

9.3 SO_REUSEADDR vs SO_REUSEPORT

选项作用应用场景
SO_REUSEADDR允许绑定处于 TIME_WAIT 的地址；多 socket 绑不同 IP 同端口服务快速重启；通配+具体IP 共存
SO_REUSEPORT (Linux 3.9+)多 socket 绑完全相同 IP:Port，内核哈希分发多 worker 无锁 accept，消除惊群
// 现代高性能服务器：每个 worker 独立 listen
for (int i = 0; i < num_workers; ++i) {
    if (fork() == 0) {
        int fd = socket(AF_INET, SOCK_STREAM, 0);
        int one = 1;
        setsockopt(fd, SOL_SOCKET, SO_REUSEPORT, &one, sizeof(one));
        bind(fd, ...);    // 多个进程绑同一端口
        listen(fd, 1024);
        acceptLoop(fd);   // 内核哈希分发，负载均衡
    }
}
// Nginx 1.9.1+、HAProxy 均使用 SO_REUSEPORT

9.6 io_uring — Linux 5.1+ 异步 I/O 革命

传统 epoll 是就绪通知 + 用户态 syscall；io_uring 通过用户/内核共享的环形队列，提交和完成 I/O 几乎零系统调用。

// liburing 简化 API
#include <liburing.h>
struct io_uring ring;
io_uring_queue_init(256, &ring, 0);

// 提交异步 read
auto* sqe = io_uring_get_sqe(&ring);
io_uring_prep_read(sqe, fd, buf, 4096, 0);
io_uring_submit(&ring);

// 获取完成事件
struct io_uring_cqe* cqe;
io_uring_wait_cqe(&ring, &cqe);
io_uring_cqe_seen(&ring, cqe);

性能：高并发小 I/O 比 epoll 快 30-200%。TigerBeetle、Seastar、tokio-uring 已采用。

9.8 高频面试题 Q&A

Q1: accept 返回的 fd 跟 listenfd 有什么区别？

listenfd 只负责接收连接（管 SYN 队列、accept 队列）；connfd 代表已建立的连接，有独立四元组和收发缓冲区。多个 connfd 共享同一 listenfd，靠四元组区分。

Q2: TCP 为什么可靠？HTTP/3 为什么换用 UDP？

TCP 可靠机制：序列号+确认、超时重传、滑动窗口、流控、拥塞控制、校验和。

HTTP/3 换用 QUIC：TCP 的队头阻塞——一包丢失阻塞所有流。QUIC 基于 UDP 自实现可靠性，每流独立重传，切网络不断连，0-RTT 握手。

Q3: 惊群效应是什么？怎么解决？

惊群：多个进程 accept 同一 listenfd，连接到来时全部被唤醒，但只有一个成功。

  • Linux 2.6 已修复 accept 惊群
  • epoll_wait 惊群靠 EPOLLEXCLUSIVE（4.5+）
  • 根治：SO_REUSEPORT — 每 worker 独立 listenfd，内核哈希分发
Q4: C10K / C10M 问题？

C10K：单机 1 万并发。epoll + 非阻塞 + Reactor 已解决。

C10M：单机 1000 万。需 bypass 内核栈（DPDK、用户态 TCP mTCP/F-Stack/Seastar）+ 零拷贝 + 大页 + NUMA 亲和 + 多队列网卡 + RSS + per-CPU 数据结构。

Q5: epoll 为什么高效？（底层原理）
  • select/poll：内核遍历所有 fd 检查状态 → O(n)
  • epoll：fd 注册时把回调挂到设备等待队列，事件发生时回调把 fd 入就绪链表 → epoll_wait 只读链表，O(1)

这就是"事件驱动"——内核主动通知而非轮询。

Q6: 大端 vs 小端？什么时候要关心？

x86/ARM 普遍小端（低字节在低地址），网络字节序是大端。数据跨主机/跨架构时必须转换：htons/htonl、ntohs/ntohl。Protobuf 内部已处理；手写二进制协议（如 Adapter<T> + memcpy）时必须警惕。

Q7: HTTPS 握手做了什么？性能优化？

TLS 1.2：2-RTT 握手；TLS 1.3：1-RTT（首次）、0-RTT（恢复）。

优化：Session Resumption、OCSP Stapling、HTTP/2 连接复用、ECDHE、硬件加速（AES-NI、Intel QAT）。

Q8: 如何设计一个百万连接的长连接服务器？
  • 架构：多进程 per-core worker + SO_REUSEPORT；每 worker 一个 epoll Reactor；I/O 与业务线程分离；连接用哈希表 + 时间轮超时
  • 内核参数：ulimit -n 1048576、somaxconn、tcp_max_syn_backlog、tcp_tw_reuse、ip_local_port_range
  • 内存：每连接 ~10KB → 100 万约 10-20 GB
  • 心跳：应用层 30s + TCP keepalive 兜底 + 时间轮 O(1) 清理
