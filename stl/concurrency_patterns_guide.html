3.3 promise + future 经典模式

// 工作线程把结果通过 promise 传给主线程
std::promise<int> p;
std::future<int> f = p.get_future();

std::thread t([p = std::move(p)]() mutable {
    try {
        int result = heavy_work();
        p.set_value(result);
    } catch (...) {
        p.set_exception(std::current_exception());        // 异常也传过去
    }
});

try {
    int r = f.get();        // 阻塞等待 — 异常会被重抛
    use(r);
} catch (const std::exception& e) {
    LOG("failed: {}", e.what());
}
t.join();

3.4 std::async — 一行启动异步

// 启动策略 - 重要细节
auto f1 = std::async(std::launch::async,    work);   // 立刻新线程
auto f2 = std::async(std::launch::deferred, work);   // 同步执行 (在 get/wait 时)
auto f3 = std::async(work);                            // ⚠️ 由实现选择 — 不可控

// 默认策略不可控 → 推荐显式 std::launch::async
std::async 的两个大坑：
  1. 不指定策略时，"async | deferred" 由实现决定 → 可能根本没新线程，不并发
  2. std::async 返回的 future 析构时会等待任务完成(隐式 join 风格) → 看似并行，实际逐个完成。
    std::async(work1); std::async(work2); — 两个 future 都是临时对象 → 立即析构 → 串行! 
// ❌ 可能串行执行 (future 临时对象立即析构等待任务完成)
std::async(std::launch::async, work1);
std::async(std::launch::async, work2);

// ✅ 并行
auto f1 = std::async(std::launch::async, work1);
auto f2 = std::async(std::launch::async, work2);
f1.get(); f2.get();

3.5 packaged_task — 让任意函数变成 future-able 的任务

std::packaged_task<int(int)> task([](int x){ return x * x; });
std::future<int> f = task.get_future();

// 投递到线程池或裸线程
std::thread{std::move(task), 5}.detach();

int r = f.get();      // 25
// packaged_task 是线程池 submit 的核心组件

3.6 shared_future — 多个等待者

std::promise<Config> p;
std::shared_future<Config> sf = p.get_future().share();

// 多个 worker 都等同一个配置加载完成
for (int i = 0; i < 10; ++i) {
    std::thread{[sf]{ auto cfg = sf.get(); use(cfg); }}.detach();
}

// 主线程加载好后通知所有 worker
p.set_value(load_config());
"future / promise 怎么实现?" — 答：内部有共享状态（堆分配，含 mutex + cv + value/exception 槽 + ready 标志）。promise.set_value() 加锁、写值、置 ready、notify_all；future.get() 加锁、cv.wait(ready)、返回值或抛异常。引用计数管理共享状态生命周期。

std::shared_mutex 读写锁 (C++17)

用法 · 性能 — 反直觉 · 锁升级 / 降级 — 标准不支持 · 何时不要用 shared_mutex

4.1 用法

class SessionTable {
    std::shared_mutex mu_;
    std::unordered_map<SessionId, std::shared_ptr<Session>> m_;
public:
    std::shared_ptr<Session> find(SessionId id) const {
        std::shared_lock lk{mu_};               // 读锁 — 多个线程可同时持有
        auto it = m_.find(id);
        return it == m_.end() ? nullptr : it->second;
    }
    void insert(SessionId id, std::shared_ptr<Session> s) {
        std::unique_lock lk{mu_};               // 写锁 — 独占
        m_.emplace(id, std::move(s));
    }
    void erase(SessionId id) {
        std::unique_lock lk{mu_};
        m_.erase(id);
    }
};

4.2 性能 — 反直觉

读写锁不一定比 mutex 快！

场景shared_mutexmutex结论
读多写少 (90% 读)快慢shared_mutex 胜
读写各半≈≈差不多
读多但临界区极短慢快mutex 胜（shared_mutex 自身开销大）
大量并发读快慢shared_mutex 大胜

原因：shared_mutex 实现复杂，本身的 lock/unlock 开销 ~5× 普通 mutex。临界区极短时，"开销 > 节省的等待时间"。

4.3 锁升级 / 降级 — 标准不支持

// ❌ C++ 标准不支持锁升级 (read → write 升级)
std::shared_lock lk{mu_};
if (need_modify()) {
    // 必须先 unlock 再 unique_lock — 中间窗口可能其他线程改动数据
    lk.unlock();
    std::unique_lock wl{mu_};
    // 重新检查条件 (类似 DCLP)
    if (still_need_modify()) do_modify();
}
升级陷阱：两个线程同时持读锁 + 都想升级 → 死锁。所以 C++ 标准故意不提供升级接口。如果业务需要先读再写，一开始就拿写锁。

4.4 何时不要用 shared_mutex

  • 读临界区极短 (一个变量读取) → 直接 atomic
  • 读写比例接近 1:1 → 用普通 mutex
  • 对延迟敏感 → 用 RCU (Read-Copy-Update) 或 hazard pointer
  • 需要锁升级 → 重新设计 (单写者模式 / 单 writer queue)

6.1 V1 — 最简线程池（不能返结果）

class SimplePool {
    std::vector<std::thread> workers_;
    std::queue<std::function<void()>> tasks_;
    std::mutex mu_;
    std::condition_variable cv_;
    bool stop_ = false;
public:
    explicit SimplePool(size_t n) {
        for (size_t i = 0; i < n; ++i)
            workers_.emplace_back([this]{ loop(); });
    }
    ~SimplePool() {
        { std::lock_guard lk{mu_}; stop_ = true; }
        cv_.notify_all();
        for (auto& t : workers_) t.join();
    }
    void submit(std::function<void()> f) {
        { std::lock_guard lk{mu_}; tasks_.emplace(std::move(f)); }
        cv_.notify_one();
    }
private:
    void loop() {
        while (true) {
            std::function<void()> task;
            {
                std::unique_lock lk{mu_};
                cv_.wait(lk, [this]{ return stop_ || !tasks_.empty(); });
                if (stop_ && tasks_.empty()) return;
                task = std::move(tasks_.front());
                tasks_.pop();
            }
            task();
        }
    }
};

6.2 V2 — 支持返回值 + 异常透传 (生产级)

class ThreadPool {
    std::vector<std::jthread> workers_;
    std::queue<std::move_only_function<void()>> tasks_;        // C++23 (或 std::function)
    std::mutex mu_;
    std::condition_variable_any cv_;

public:
    explicit ThreadPool(size_t n = std::thread::hardware_concurrency()) {
        workers_.reserve(n);
        for (size_t i = 0; i < n; ++i)
            workers_.emplace_back([this](std::stop_token st){ loop(st); });
    }

    template<class F, class... Args>
    auto submit(F&& f, Args&&... args)
        -> std::future<std::invoke_result_t<F, Args...>>
    {
        using R = std::invoke_result_t<F, Args...>;

        // packaged_task 把返回值/异常都包进 future
        auto task = std::make_shared<std::packaged_task<R()>>(
            [f = std::forward<F>(f),
             args = std::make_tuple(std::forward<Args>(args)...)]() mutable {
                return std::apply(std::move(f), std::move(args));
            });

        std::future<R> fut = task->get_future();
        {
            std::lock_guard lk{mu_};
            tasks_.emplace([task]{ (*task)(); });
        }
        cv_.notify_one();
        return fut;
    }

    void stop() {
        for (auto& w : workers_) w.request_stop();
        cv_.notify_all();
    }

private:
    void loop(std::stop_token st) {
        std::stop_callback cb(st, [this]{ cv_.notify_all(); });
        while (!st.stop_requested()) {
            std::move_only_function<void()> task;
            {
                std::unique_lock lk{mu_};
                cv_.wait(lk, st, [this]{ return !tasks_.empty(); });
                if (st.stop_requested() && tasks_.empty()) return;
                task = std::move(tasks_.front());
                tasks_.pop();
            }
            try { task(); }
            catch (...) { /* packaged_task 内部已吞 — 异常存到 future */ }
        }
    }
};

// 使用 — 任意 callable + 任意参数, 自动获得 future
ThreadPool pool;
auto f1 = pool.submit([](int a, int b){ return a + b; }, 1, 2);
auto f2 = pool.submit([]{ throw std::runtime_error{"oops"}; });

int r = f1.get();             // = 3
try { f2.get(); }
catch (const std::exception& e) { LOG("task failed: {}", e.what()); }

6.4 线程池大小选择公式

经典 (Brian Goetz)：

N_threads = N_cpu * U_cpu * (1 + W/C)
其中:
  N_cpu = CPU 核数
  U_cpu = 目标 CPU 利用率 (0~1)
  W/C   = 等待时间 / 计算时间 比值
  • CPU 密集（W/C ≈ 0）→ 线程数 ≈ N_cpu
  • IO 密集（W/C 高，比如 10）→ 线程数 ≈ N_cpu × 10+
  • 实际：用监控数据调优 (任务延迟 / 队列长度 / CPU 使用率)

7.1 BoundedQueue（生产级实现）

template<class T>
class BoundedQueue {
    std::queue<T> q_;
    std::mutex mu_;
    std::condition_variable cv_not_full_, cv_not_empty_;
    size_t cap_;
    bool closed_ = false;
public:
    explicit BoundedQueue(size_t cap) : cap_(cap) {}

    // 阻塞 push — 满了等
    bool push(T v) {
        std::unique_lock lk{mu_};
        cv_not_full_.wait(lk, [this]{ return closed_ || q_.size() < cap_; });
        if (closed_) return false;
        q_.push(std::move(v));
        cv_not_empty_.notify_one();
        return true;
    }

    // 带超时的 push
    bool push_for(T v, std::chrono::milliseconds tm) {
        std::unique_lock lk{mu_};
        if (!cv_not_full_.wait_for(lk, tm, [this]{
            return closed_ || q_.size() < cap_;
        })) return false;
        if (closed_) return false;
        q_.push(std::move(v));
        cv_not_empty_.notify_one();
        return true;
    }

    // 阻塞 pop — 关闭且空时返回 nullopt
    std::optional<T> pop() {
        std::unique_lock lk{mu_};
        cv_not_empty_.wait(lk, [this]{ return closed_ || !q_.empty(); });
        if (q_.empty()) return std::nullopt;       // closed + drained
        T v = std::move(q_.front()); q_.pop();
        cv_not_full_.notify_one();
        return v;
    }

    void close() {
        std::lock_guard lk{mu_};
        closed_ = true;
        cv_not_full_.notify_all();
        cv_not_empty_.notify_all();
    }

    size_t size() const { std::lock_guard lk{mu_}; return q_.size(); }
};

class RouteTable {

8.2 Copy-on-Write — 极致读性能

"读永不阻塞，写换整个数据结构"。读路径只是原子加载一个 shared_ptr → 几 ns。

class RouteTableCOW {
    using Map = std::unordered_map<std::string, Handler>;
    std::atomic<std::shared_ptr<const Map>> current_;
    std::mutex write_mu_;        // 仅写者间互斥
public:
    RouteTableCOW() : current_(std::make_shared<Map>()) {}

    Handler find(const std::string& path) const {
        auto snapshot = current_.load();         // 原子加载, 几 ns
        auto it = snapshot->find(path);
        return it == snapshot->end() ? nullptr : it->second;
    }

    void add(const std::string& path, Handler h) {
        std::lock_guard lk{write_mu_};
        auto snap = current_.load();
        auto next = std::make_shared<Map>(*snap);   // 复制!
        (*next)[path] = std::move(h);
        current_.store(next);                       // 原子替换
    }
};

适用场景：读 ≫ 写（路由表、配置、metadata）。不适用大数据集（每次写都全量复制）。

Pipeline 流水线模式

实现骨架 · 性能调优要点

把任务分阶段，每阶段一个/多个线程，阶段间用队列连接 → 高吞吐 + 资源隔离。

9.1 实现骨架

template<class In, class Out>
class Stage {
    BoundedQueue<In>& in_;
    BoundedQueue<Out>& out_;
    std::function<Out(In)> transform_;
    std::vector<std::jthread> workers_;
public:
    Stage(BoundedQueue<In>& in, BoundedQueue<Out>& out,
          std::function<Out(In)> t, size_t n_workers)
        : in_(in), out_(out), transform_(std::move(t))
    {
        for (size_t i = 0; i < n_workers; ++i)
            workers_.emplace_back([this](std::stop_token st){ loop(st); });
    }
private:
    void loop(std::stop_token st) {
        while (!st.stop_requested()) {
            auto item = in_.pop();
            if (!item) {                            // 上游关闭
                out_.close();
                return;
            }
            try {
                out_.push(transform_(std::move(*item)));
            } catch (const std::exception& e) {
                LOG_ERROR("stage error: {}", e.what());
            }
        }
    }
};

// 组装 4 阶段流水线
BoundedQueue<RawBuf>  q1{1024};
BoundedQueue<Parsed>   q2{1024};
BoundedQueue<Valid>    q3{1024};
BoundedQueue<Output>   q4{1024};

Stage<RawBuf, Parsed> s1{q1, q2, parse,     2};
Stage<Parsed, Valid>  s2{q2, q3, validate,  4};   // 瓶颈阶段多 worker
Stage<Valid, Output>  s3{q3, q4, transform, 2};
Stage<Output, void>   s4{q4, ..., write,    1};

9.2 性能调优要点

  • 找瓶颈阶段：监控每阶段队列长度 — 长的就是瓶颈
  • 瓶颈阶段加 worker（如果可并行）
  • 批处理：每次 pop N 个一起处理（减少 cv 唤醒次数）
  • 零拷贝：用 shared_ptr 在阶段间传递大对象
  • 避免假共享：不同阶段的队列对象间隔 cache line

10.1 Active Object 实现

class Counter {
    std::queue<std::function<void()>> mailbox_;
    std::mutex mu_;
    std::condition_variable cv_;
    std::jthread worker_;
    int count_ = 0;       // 只在 worker 线程内修改

public:
    Counter() : worker_([this](std::stop_token st){ loop(st); }) {}

    // 投递消息 → 立即返回 future
    std::future<int> increment() {
        return post([this]{ return ++count_; });
    }
    std::future<int> get() {
        return post([this]{ return count_; });
    }

private:
    template<class F>
    auto post(F f) -> std::future<std::invoke_result_t<F>> {
        using R = std::invoke_result_t<F>;
        auto task = std::make_shared<std::packaged_task<R()>>(std::move(f));
        auto fut = task->get_future();
        { std::lock_guard lk{mu_}; mailbox_.emplace([task]{ (*task)(); }); }
        cv_.notify_one();
        return fut;
    }

    void loop(std::stop_token st) {
        std::stop_callback cb(st, [this]{ cv_.notify_all(); });
        while (!st.stop_requested()) {
            std::function<void()> msg;
            {
                std::unique_lock lk{mu_};
                cv_.wait(lk, st, [this]{ return !mailbox_.empty(); });
                if (mailbox_.empty()) return;
                msg = std::move(mailbox_.front()); mailbox_.pop();
            }
            msg();
        }
    }
};

// 使用 — 完全无锁的多线程访问
Counter c;
std::vector<std::future<int>> results;
for (int i = 0; i < 1000; ++i)
    results.emplace_back(c.increment());
for (auto& f : results) f.wait();

Work-Stealing 工作窃取

动机 · 关键设计 · 工程框架

11.1 动机

普通线程池一个全局任务队列 → 高并发下队列锁是瓶颈。

Work-Stealing：每个 worker 自己的队列 (deque)，自己 push/pop 自己的尾部；空了去偷别人队列的头。

11.2 关键设计

  • 每个 worker 持有 双端队列
  • 本 worker 从 deque 尾 push/pop（栈式，缓存友好）
  • 空闲 worker 从其他人 deque 头 偷（FIFO 减少冲突）
  • 双端队列内部用原子操作实现 lock-free（Chase-Lev deque）

11.3 工程框架

  • Intel TBB：成熟工业级
  • folly::CPUThreadPoolExecutor：Facebook 实现
  • std::execution::par 后端：标准并行算法
  • 自实现：从零写 Chase-Lev deque 难度极大，建议用现成库
// 概念示意 — 实际生产用 TBB / folly
class WorkStealingPool {
    std::vector<WorkerQueue> queues_;        // 每 worker 一个 lock-free deque
    std::vector<std::jthread> workers_;
    std::atomic<size_t> next_submit_{0};
public:
    void submit(std::function<void()> f) {
        auto idx = next_submit_.fetch_add(1) % queues_.size();
        queues_[idx].push_back(std::move(f));    // round-robin 投递
    }
private:
    void worker_loop(size_t id, std::stop_token st) {
        auto& my_q = queues_[id];
        while (!st.stop_requested()) {
            auto task = my_q.pop_back();          // 自己的尾, lock-free
            if (task) { (*task)(); continue; }

            // 偷别人
            for (size_t i = 0; i < queues_.size(); ++i) {
                if (i == id) continue;
                auto stolen = queues_[i].pop_front();   // 别人的头
                if (stolen) { (*stolen)(); break; }
            }
            // 都没活 — sleep / spin / wait
        }
    }
};
核心收益：① 大多数 push/pop 是 lock-free；② 缓存友好（栈式访问 + 偷别人是少数事件）；③ 自动负载均衡。

11b.1 来历与定位

LMAX 是 2010 年代英国一家外汇交易所，单线程 600 万订单/秒 的吞吐，靠的不是更多线程，而是一种被命名为 Disruptor 的并发模式。核心论文：Disruptor: High Performance Alternative To Bounded Queues for Exchanging Data Between Concurrent Threads。它的设计哲学叫 Mechanical Sympathy —— 顺应 CPU 硬件特性。

Disruptor 不是"更快的队列"，而是"一种重新定义的多写多读、保留顺序、无锁回收的环形 buffer 协议"。在低延时领域（金融撮合、做市、行情广播、游戏服务器、Aeron）是事实标准。

11b.4 简化实现（200 行思路）

// disruptor_lite.hpp - 把核心思想写成可跑代码
template<class Event, size_t N>
class RingBuffer {
    static_assert((N & (N - 1)) == 0, "N must be power of 2");
    Event slots_[N];
    static constexpr size_t MASK = N - 1;
public:
    Event& at(int64_t seq) { return slots_[seq & MASK]; }
};

class Sequence {                          // 每个 cursor 独占一个 cache line
    alignas(64) std::atomic<int64_t> value_{-1};
    char _pad[64 - sizeof(std::atomic<int64_t>)];
public:
    int64_t get() const { return value_.load(std::memory_order_acquire); }
    void set(int64_t v) { value_.store(v, std::memory_order_release); }
};

template<class Event, size_t N>
class Disruptor {
    RingBuffer<Event, N>     ring_;
    Sequence                  producer_;       // 生产者已申请到的最大 seq
    Sequence                  published_;      // 已发布的最大 seq
    std::vector<Sequence*>    consumers_;      // 所有消费者的 cursor

public:
    // 生产者：申请 slot
    int64_t next() {
        int64_t n = producer_.get() + 1;
        // 等到所有消费者都不再用 slot (n - N)
        while (n - (int64_t)N > min_consumer_seq()) {
            std::this_thread::yield();         // 真实库用 WaitStrategy
        }
        producer_.set(n);
        return n;
    }

    Event& get(int64_t seq) { return ring_.at(seq); }

    // 生产者：发布
    void publish(int64_t seq) { published_.set(seq); }

    // 消费者：等到 seq 可读，返回当前最大可读 seq（批量！）
    int64_t wait_for(int64_t seq) {
        int64_t avail;
        while ((avail = published_.get()) < seq) {
            std::this_thread::yield();
        }
        return avail;
    }
private:
    int64_t min_consumer_seq() {
        int64_t m = INT64_MAX;
        for (auto* c : consumers_) m = std::min(m, c->get());
        return m;
    }
};
// 使用 — 单生产者 + 两个串行消费者（日志 + 持久化）
Disruptor<Order, 8192> d;
Sequence logger_cursor, persist_cursor;
d.consumers().push_back(&logger_cursor);
d.consumers().push_back(&persist_cursor);

// 生产者线程（行情接入）
for (;;) {
    Order o = read_from_socket();
    int64_t seq = d.next();
    d.get(seq) = std::move(o);    // 直接写到 ring 槽位 — 零拷贝
    d.publish(seq);
}

// 消费者线程
int64_t cur = 0;
for (;;) {
    int64_t end = d.wait_for(cur);     // 批量到达
    for (int64_t i = cur; i <= end; ++i) {
        handle(d.get(i));         // 处理 (i - cur + 1) 个事件，cache 友好
    }
    logger_cursor.set(end);
    cur = end + 1;
}

机制带来的收益
预分配 + 复用 slot运行时零 malloc → 无锁分配器的争用
每个 cursor 独占 cache line生产者写 producer_cursor 不会 invalidate 消费者 cursor → 无假共享
消费者批量拉取一次 wait_for 返回 N 个；分摊系统调用 / 内存屏障开销

面试 Q：你说 mutex 队列 enqueue 是 100ns 级，Vyukov MPMC 是 30ns，Disruptor 是 10ns —— 它究竟省了什么？
答：① 没有节点分配；② 生产者不和消费者抢同一个 atomic（各有 cursor）；③ 消费者批量摊销同步开销；④ slot 复用避开 cache 冷启动；⑤ 等待策略可调到 BusySpin 跳过 OS 调度。综合下来一次 enqueue 平均只剩 1 次 atomic store + 1 次 cache 写入。

11b.6 C++ 生态

  • Disruptor-cpp — 最完整的 C++ 移植，金融行业常用
  • Aeron — Disruptor 作者 Martin Thompson 后来做的跨进程 / 跨主机版本，基于共享内存 + UDP 多播
  • moodycamel::ConcurrentQueue — 不是 Disruptor，但同样追求高吞吐，单消息语义而非广播

11c.1 它要解决什么

过去 15 年异步 C++ 的痛点循环出现：

  • Callback hell → 引入 std::future → 但 future 不能组合（没有 then / when_all 在标准里）
  • 第三方 future（folly、Boost.Future）各成生态、互不兼容
  • C++20 协程 → 解决了"线性异步代码可读性"，但调度器、取消、组合仍是裸的
  • 没有标准的异步执行抽象 → 每个库（Asio、TBB、CUDA、GPU）自己一套

P2300（std::execution）由 NVIDIA、Microsoft、Meta、Google 联合推动，进入 C++26。它定义一套通用的"异步操作 + 调度器 + 结构化取消"代数，标准库、Asio、CUDA、SyCL、Vulkan 都将基于这套接口。

11c.2 三大概念

概念类比职责
Scheduler"在哪里跑"线程池、IO 线程、GPU、协程框架的统一抽象
Sender"做什么"（懒计算）异步操作的描述符；不调度也不执行，可被组合
Receiver"完成后给谁"三个回调入口：set_value / set_error / set_stopped

11c.7 结构化并发 (Structured Concurrency) — 取消、生命周期、所有权

Sender/Receiver 不只是"另一种 future"，它强制 结构化并发：

  • 子任务不能比父任务活得久：父结束时所有未完成子 sender 自动收到 set_stopped
  • 取消是头等公民：stop_token 由 receiver 提供，沿组合链自动传播，无需手动埋点
  • 无悬空回调：sender 不能"逃出"作用域（不像 std::thread::detach）
  • 无 race-on-error：when_all 任意一个失败 → 其他自动取消，全部等齐再上报
这就是 Trio / Kotlin Coroutines / Swift Structured Concurrency 引领过的范式，C++26 终于跟上。它从根本上消灭一类异步代码 bug：goroutine / detached thread / dangling callback 泄漏。

12.3 简化 Task 实现骨架

template<class T>
class Task {
public:
    struct promise_type {
        std::optional<T> value;
        std::exception_ptr eptr;

        Task get_return_object() {
            return Task{std::coroutine_handle<promise_type>::from_promise(*this)};
        }
        std::suspend_never initial_suspend() noexcept { return {}; }
        std::suspend_always final_suspend() noexcept { return {}; }
        void return_value(T v) { value = std::move(v); }
        void unhandled_exception() { eptr = std::current_exception(); }
    };

    std::coroutine_handle<promise_type> h_;

    // awaiter — 让 co_await Task 工作
    bool await_ready() const { return h_.done(); }
    void await_suspend(std::coroutine_handle<> awaiting) { /* 调度 */ }
    T await_resume() {
        auto& p = h_.promise();
        if (p.eptr) std::rethrow_exception(p.eptr);
        return std::move(*p.value);
    }
};

        catch (...) { errs.push_back(std::current_exception()); }

}

死锁的预防与检测

14.1 6 条防御规则

  1. 锁的全局顺序：所有线程按同一顺序获取锁 (例如按地址排序)
  2. 同时获取多锁用 std::lock / std::scoped_lock (C++17)
  3. 锁的粒度尽量小：只锁数据，不锁 IO/计算
  4. RAII 锁守卫：lock_guard / unique_lock，永远不裸 lock/unlock
  5. 不在持锁时调用未知代码（用户回调、虚函数）
  6. 不持锁时阻塞等待（不在持锁时 cv.wait 之外的等待）

14.2 同时锁多个 mutex 的正确写法

// ❌ 错误 - 顺序不一致 → 死锁
std::mutex a, b;
void thread1() { std::lock_guard la{a}; std::lock_guard lb{b}; }
void thread2() { std::lock_guard lb{b}; std::lock_guard la{a}; }    // 反向

// ✅ C++17 scoped_lock - 内部用 deadlock-avoiding 算法
void thread1() { std::scoped_lock l{a, b}; }
void thread2() { std::scoped_lock l{b, a}; }    // 同样 OK

// 旧 C++11 写法
std::lock(a, b);
std::lock_guard la{a, std::adopt_lock};
std::lock_guard lb{b, std::adopt_lock};

14.3 锁层次 (Lock Hierarchy)

class HierarchicalMutex {
    std::mutex internal_;
    const uint64_t level_;
    static thread_local uint64_t current_level_;
    uint64_t prev_level_;
public:
    explicit HierarchicalMutex(uint64_t lvl) : level_(lvl) {}

    void lock() {
        if (current_level_ <= level_)
            throw std::logic_error{"violates hierarchy"};
        internal_.lock();
        prev_level_ = current_level_;
        current_level_ = level_;
    }
    void unlock() {
        current_level_ = prev_level_;
        internal_.unlock();
    }
};
thread_local uint64_t HierarchicalMutex::current_level_ = UINT64_MAX;

// 设定层次：高层锁先持有, 低层锁后持有
HierarchicalMutex route_mu{1000};        // 高层
HierarchicalMutex stats_mu{100};         // 低层
// 强制 route_mu 先于 stats_mu — 反序时编译/运行报错

15.1 platform agent — Reactor + 线程池组合

class PlatformAgent {
    EventLoop loop_;                          // 单线程 epoll, 处理 IO
    ThreadPool worker_pool_{8};               // 业务计算 8 线程
    ThreadPool db_pool_{4};                   // DB 调用 4 线程

public:
    void on_request(std::shared_ptr<Session> sess, RpcRequest req) {
        // IO 线程 → 业务线程池 (避免阻塞 epoll)
        worker_pool_.submit([sess, req = std::move(req)]() mutable {
            try {
                auto response = handle_rpc(req);
                // 回到 IO 线程发送响应 (避免多线程访问 socket)
                sess->post_to_io([sess, response]{ sess->send(response); });
            } catch (const std::exception& e) {
                sess->post_to_io([sess, msg = std::string{e.what()}]{
                    sess->send_error(500, msg);
                });
            }
        });
    }
};

15.2 异步日志 — Producer-Consumer + batch flush

class AsyncLogger {
    BoundedQueue<LogEntry> q_{16384};
    std::jthread flusher_;
public:
    AsyncLogger(const std::string& path) {
        flusher_ = std::jthread([this, path](std::stop_token st){
            std::ofstream f{path};
            std::vector<LogEntry> batch;
            batch.reserve(256);
            while (!st.stop_requested()) {
                // 批量取 — 减少 cv 唤醒次数
                if (auto e = q_.pop()) {
                    batch.push_back(std::move(*e));
                    while (batch.size() < 256) {
                        auto e2 = q_.try_pop();   // 非阻塞
                        if (!e2) break;
                        batch.push_back(std::move(*e2));
                    }
                    for (const auto& e : batch) f << e.format() << '\n';
                    f.flush();
                    batch.clear();
                }
            }
        });
    }
    void log(LogEntry e) { q_.push(std::move(e)); }
};

15.4 限流器 — 令牌桶 (atomic)

class TokenBucket {
    const double rate_;          // tokens / sec
    const double capacity_;
    std::atomic<double> tokens_;
    std::atomic<std::chrono::steady_clock::time_point> last_;
public:
    TokenBucket(double r, double c)
        : rate_(r), capacity_(c), tokens_(c), last_(std::chrono::steady_clock::now()) {}

    bool try_acquire(double n = 1) {
        auto now = std::chrono::steady_clock::now();
        auto prev = last_.exchange(now);
        double dt = std::chrono::duration<double>(now - prev).count();
        double avail = std::min(capacity_, tokens_.load() + dt * rate_);
        if (avail < n) {
            tokens_.store(avail);
            return false;
        }
        tokens_.store(avail - n);
        return true;
    }
};
// 注: 这是简化版 - 严格的并发安全令牌桶需要 CAS 循环

15.5 优雅停服

int main() {
    Server server;
    server.start();

    // 信号 → atomic flag → 主循环检测
    static std::atomic<bool> stopping{false};
    signal(SIGTERM, [](int){ stopping = true; });

    while (!stopping.load()) std::this_thread::sleep_for(100ms);

    // 优雅停服 4 步
    server.stop_accepting_new();              // 1. 停止接受新连接
    server.drain_pending(30s);                // 2. 处理完未完任务 (有超时)
    server.close_connections();              // 3. 关连接
    // 4. server 析构 → jthread 自动 stop + join
}

易错点速查（10 个高频陷阱）

  1. std::async 临时 future 串行 
    std::async(work1); std::async(work2);     // ⚠️ 串行!
auto f1 = std::async(...); auto f2 = std::async(...);    // ✅ 并行
  2. cv.wait 不带谓词 → 虚假唤醒 
    cv.wait(lk);                   // ⚠️ 虚假唤醒不退
cv.wait(lk, [&]{ return ready; });    // ✅
  3. 持锁调用回调 → 死锁 
    std::lock_guard lk{mu_};
user_callback();          // ⚠️ 用户回调可能拿其他锁 → 死锁
  4. thread 析构没 join → terminate 
    { std::thread t{work}; }    // ⚠️ 析构 abort()
{ std::jthread t{work}; }   // ✅ C++20 自动 join
  5. 引用捕获跨线程 → 悬空 
    {
    int x = 42;
    pool.submit([&]{ use(x); });    // ⚠️ x 可能已销毁
}
  6. future.get() 多次 → terminate
    future 只能 get 一次。多线程共享要用 shared_future。
  7. 持锁时间过长
    锁内做 IO / sleep / 大计算 → 其他线程长时间等待。
  8. shared_mutex 临界区太短
    不如普通 mutex 快 (shared_mutex 自身开销大)。
  9. 线程数过多
    上下文切换开销 ≫ 计算收益。线程数 ≈ N_cpu × (1 + W/C)。
  10. atomic 误用 memory_order_relaxed
    不能保证 happens-before，导致同步失效。详见 cpp_memory_model.html。
